#### 上传服务器后的权限执行

# 设置项目目录所有权为www-data（Nginx运行用户）
sudo chown -R www-data:www-data /var/www/mj-gallery

# 设置正确的目录权限
sudo find /var/www/mj-gallery -type d -exec chmod 755 {} \;

# 设置正确的文件权限
sudo find /var/www/mj-gallery -type f -exec chmod 644 {} \;

# 对于需要执行权限的脚本（如果有）
sudo chmod +x /var/www/mj-gallery/server/index.js
